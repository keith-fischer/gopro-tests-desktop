def script_init():
    gda_utils.GetEnvInfo()
    app_gda, region_gda = gda_utils.AppStart("Quik",0,0)

    if app_gda and region_gda:
        print("target region is ready")
        return app_gda,region_gda
    print "Failed to init Quik region"
    exit(-1)
