def create_your_acct(REGION,willfail=True,fname="john",lname="doe",email="jdoe@jdoe.zzz",pw="1234567890"):
    gda_utils.WAIT(REGION,Pattern().exact()) #createacct_txt_title.png

    gda_utils.FIND(REGION,Pattern().similar(0.90)) #createacct_txt_subtitle.png

    gda_utils.TYPE(REGION,Pattern().exact().targetOffset(-232,23),fname) #createacct_txtbox_email.png
    gda_utils.TYPE(REGION,Pattern().exact().targetOffset(51,26),lname) #createacct_txtbox_email.png
    gda_utils.TYPE(REGION,Pattern().exact().targetOffset(-245,-32),email) #createacct_txtbox_email.png
    gda_utils.TYPE(REGION,Pattern().exact().targetOffset(-216,-29),pw) #createacct_txtbox_password.png
    gda_utils.TYPE(REGION,Pattern().exact().targetOffset(-247,28),pw) #createacct_txtbox_password.png

    if gda_utils.d_gda_settings["isWindows"]=="True":
        gda_utils.CLICK(REGION,Pattern().exact().targetOffset(-240,-4)) #createacct_unchecked_getnews.png
        gda_utils.FIND(REGION,Pattern().exact().targetOffset(-246,-5)) #createacct_checked_getnews.png
        gda_utils.CLICK(REGION,Pattern().similar(0.98).targetOffset(-275,-7)) #createacct_unchecked_iacknowledge.png
        gda_utils.FIND(REGION,Pattern().similar(0.91).targetOffset(-272,-2)) #createacct_checked_iacknowledge.png

    if gda_utils.d_gda_settings["isMac"]=="True":
        gda_utils.CLICK(REGION,Pattern().exact().targetOffset(-249,7)) #createacct_unchecked-mac_getnews.png
        gda_utils.FIND(REGION,Pattern().exact().targetOffset(-248,2)) #createacct_checked-mac_getnews.png
        gda_utils.CLICK(REGION,Pattern().similar(0.98).targetOffset(-270,-5)) #createacct_unchecked-mac_iacknowledge.png
        gda_utils.FIND(REGION,Pattern().similar(0.98).targetOffset(-271,-5)) #createacct_checked-mac_iacknowledge.png



    CLICK(REGION, Pattern().similar(0.98)) #createacct_btn_getacct.png
    wait(15) # wait for message
    #invalid acct info
    if willfail==True:
        WAIT(REGION,Pattern().exact(),30)

        CLICK(REGION,Pattern().similar(0.91)) #createacct_btn_signin.png
    else:
        CLICK(REGION,Pattern().similar(0.91)) #createacct_btn_signin.png
